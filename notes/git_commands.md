git status показывает текущее состояние рабочей директории 
git add добавляет новые изменения и файлы 
git commit сохраняет подготовленные изменения в истории, комментирует 
git push отправляет изменения в удаленный репозиторий 
git pull получает последние изменения из удаленного репозитория и объединяет с локальной веткой
git branch показывает локальные ветки 
git checkout переключает ветки
