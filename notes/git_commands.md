git status - проверка текущего нахождения в папке
git add - отслеживание изменений перенос в промежуточную память
git commit - фиксация и перенос из промежуточной памяти в локольный репохиторий
git push - отправка изменений из локального репозитория в удаленный 
git pull - стягивание изменений из удаленного репозитория в локальный
git branch - просмотр ветки
git checkout - переключение между ветками
